015    package org.codehaus.groovy.grails.plugins.springsecurity;
016    
017    import org.springframework.security.userdetails.UserDetails;
018    
019    /**
020     * Extends Spring Security's {@link UserDetails} interface to set Grails Domain
021     * Class at login, to load auth class from context.
022     *
023     * @author <a href='mailto:beckwithb@studentsonly.com'>Burt Beckwith</a>
024     */
025    public interface GrailsUser extends UserDetails {
026    
027            /**
028             * Get the domain object representing the user.
029             * @return  the user
030             */
031            Object getDomainClass();
032    }
